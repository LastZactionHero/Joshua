1 /**
  2  * Generic GameEngine Game Piece
  3  * Provides for drawing and moving a simple, game piece object
  4  * 
  5  * @constructor
  6  *
  7  * @param {int} aPieceIdx Game piece index
  8  */
  9 function GEPiece
 10 	(
 11 	aPieceIdx
 12 	)
 13 {
 14 	// Variables
 15 	this.mPieceIdx = aPieceIdx;
 16 	this.mRect = new rect( 20, 20, 40, 40 );
 17 	this.mBackgroundColor = "transparent";
 18 	this.mBackgroundImage = "url( 'img_piece.gif' )";
 19 
 20 }
 21

 23 GEPiece.prototype.constructor = GEPiece;
 24 GEPiece.prototype.baseClass = GEObject;
 25 GEPiece.prototype.generate_piece = generate_piece;
 26 GEPiece.prototype.move_piece = move_piece;
 27 GEPiece.prototype.move_piece_to_coords = move_piece_to_coords;
 28 GEPiece.prototype.get_piece_div = get_piece_div;
 29 GEPiece.prototype.get_current_region = get_current_region;
 30 GEPiece.prototype.get_center_coord = get_center_coord;
 31 GEPiece.prototype.move_to_region = move_to_region;

 33 /**
 34  * Get Game Piece Div ID
 35  * {return} Piece Div ID
 36  */
 37 function get_piece_div()
 38 {
 39 	return "d_piece_" + this.mPieceIdx;
 40 }
 41 
 42 /**
 43  * Generate Piece HTML from object properties
 44  * @return {string} Piece HTML
 45  */
 46 function generate_piece()
 47 {
 48 	return this.get_div_html( this.get_piece_div(), "" );
 49 }
 50 
 51 /**
 52  * Move piece to position in mRect
 53  */
 54 function move_piece()
 55 {
 56 	document.getElementById( this.get_piece_div() ).style.left = this.mRect.x + "px";
 57 	document.getElementById( this.get_piece_div() ).style.top = this.mRect.y + "px";
 58 	document.getElementById( this.get_piece_div() ).style.width = this.mRect.w + "px";
 59 	document.getElementById( this.get_piece_div() ).style.height = this.mRect.h + "px";
 60 }

 62 /*
 63  * Move piece to specific coordinates
 64  *
 65  * @param {coord} aCoord Desired piece coordinates
 66  */
 67 function move_piece_to_coords
 68 	(
 69 	aCoord
 70 	)
 71 {
 72 	this.mRect.x = aCoord.x;
 73 	this.mRect.y = aCoord.y;
 74 	this.move_piece();
 75 }

 77 /**
 78  * Move game piece to a specific region
 79  *
 80  * @param {int} aRegionIdx Region index
 81  */
 82 function move_to_region
 83 	(
 84 	aRegionIdx
 85 	)
 86 {
 87 	// Get rectangle of region
 88 	var regionRect = gBoard.mRegionList[aRegionIdx].mRect;
 89 
 90 	// Find coordinates in the center of the region
 91 	var regionCoords = new coord(
 92 								regionRect.x + regionRect.w / 2,
 93 								regionRect.y + regionRect.h / 2
 94 								);
 95 	regionCoords = gBoard.local_to_global_coord( regionCoords );
 96 	this.mRect.x = regionCoords.x;
 97 	this.mRect.y = regionCoords.y;
 98 			
 99 	// Move game piece
100 	this.move_piece();
101 }
102 
103 /**
104  * Get current region of game piece
105  * @return {int} Current region of game piece
106  */
107 function get_current_region()
108 {
109 	var regionCnt = gBoard.mRegionList.length;
110 	
111 	for( var regionIdx = 0; regionIdx < regionCnt; regionIdx++ )
112 	{
113 		var globalCoord = gBoard.global_to_local_coord( this.get_center_coord() );
114 			
115 		if( is_coord_in_rect( globalCoord, gBoard.mRegionList[regionIdx].mRect ) )
116 		{
117 			return regionIdx;
118 		}
119 	}
120 	return -1;
121 }
122 
123 /**
124  * Get center coordinate of game piece
125  * @return {coord} Center coordinate of game piece
126  */
127 function get_center_coord()
128 {
129 	var centerX = this.mRect.x + this.mRect.w / 2;
130 	var centerY = this.mRect.y + this.mRect.h / 2;
131 	
132 	return new coord( centerX, centerY );
133 }

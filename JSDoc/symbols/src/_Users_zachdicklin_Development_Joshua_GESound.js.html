1 /**
  2  * Generic GameEngine Sound Class
  3  * Support for initializing and playing sound effects
  4  *
  5  * @constructor
  6  * 
  7  * @param {string} aDivId Sound layer Div ID
  8  */
  9 function GESound
 10 	(
 11 	aDivId
 12 	)
 13 {
 14 	// Constants
 15 	// Sound types
 16 	SoundExtTypes = 
 17 		{ 
 18 		"SOUND_TYPE_MP3":	0,
 19 		"SOUND_TYPE_WAV":	1,
 20 		"SOUND_TYPE_NONE":	2
 21 		}
 22 		
 23 	// Sound extension values
 24 	SoundExtValues = 
 25 		[
 26 		".mp3",
 27 		".wav"
 28 		];
 29 		
 30 	// Variables
 31 	this.mDivId = aDivId;
 32 	this.audio_player_id = "id_audio_player";
 33 }
 34 
 35 GESound.prototype.play_sound_file = play_sound_file;
 36 GESound.prototype.get_file_extension = get_file_extension;
 37 GESound.prototype.draw_sound_table = draw_sound_table;
 38 GESound.prototype.get_audio_player_id = get_audio_player_id;
 39 
 40 /**
 41  * Play sound by index
 42  *
 43  * @param {int} Index of sound file
 44  */
 45 function play_sound_file
 46 	(
 47 	aIdx
 48 	)
 49 {
 50 	document.getElementById( this.get_audio_player_id( aIdx ) ).play();
 51 }
 52 
 53 /**
 54  * Get sound extension to use with this browser
 55  *
 56  * @return {string} Sound file extension
 57  */
 58 function get_file_extension()
 59 {
 60 	var ext = "";
 61 	
 62 	var browser = browser_detect();
 63 
 64 	if( browser == "Firefox" ||
 65 		browser == "Opera" )
 66 	{
 67 		ext = SoundExtValues[SoundExtTypes.SOUND_TYPE_WAV];
 68 	}
 69 	else if( browser == "Safari" ||
 70 			 browser == "Chrome" )
 71 	{
 72 		ext = SoundExtValues[SoundExtTypes.SOUND_TYPE_MP3];
 73 	}
 74 
 75 	return ext;
 76 }
 77 
 78 /**
 79  * Draw table of <audio> tags
 80  *
 81  * @param {string[]} aFilenameList List of sound filenames without extensions
 82  */
 83 function draw_sound_table
 84 	(
 85 	aFilenameList
 86 	)
 87 {
 88 	var audio_html = "";
 89 	for( var i = 0; i < aFilenameList.length; i++ )
 90 	{
 91 		var filename = aFilenameList[i] + this.get_file_extension();
 92 		audio_html += "<audio id=\"" + this.get_audio_player_id( i ) + "\" src=\"" + filename + "\" controls=\"controls\"></audio>";
 93 	}
 94 	
 95 	document.getElementById( this.mDivId ).innerHTML = audio_html;
 96 }

 99  * Get Element ID of Audio Player by Audio Id
100  *
101  * @param {int} aIdx Audio index
102  * @return {string} Audio player id
103  */
104 function get_audio_player_id
105 	(
106 	aIdx
107 	)
108 {
109 	return "audio_player_" + aIdx;
110 }

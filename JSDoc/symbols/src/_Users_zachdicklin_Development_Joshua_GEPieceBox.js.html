1 /**
  2  * Generic GameEngine Piece Box
  3  * Methods for maintaining and organizing game pieces
  4  *
  5  * @constructor
  6  *
  7  * @param {string} aDivId Game piece layer div id
  8  */
  9 function GEPieceBox
 10 	(
 11 	aDivId
 12 	)
 13 {	
 14 	// Variables: Stock Pieces
 15 	this.mPieceDiv = aDivId;
 16 	this.mPieceList = new Array();
 17 	for( i = 0; i < 10; i++ )
 18 	{
 19 		this.mPieceList[i] = new GEPiece( i );
 20 	}
 21 	
 22 	// Variables: Piece Movement
 23 	this.mSelectedPiece = null;
 24 	this.mSelectedPieceStartCoord = new coord( 0, 0 );
 25 }

 35 /**
 36  * Generate HTML for all game pieces
 37  * @return {string} HTML for all game pieces
 38  */
 39 function draw_pieces()
 40 {
 41 	var pieceHtml = "";
 42 
 43 	for( var i = 0; i < this.mPieceList.length; i++ )
 44 	{
 45 		pieceHtml += this.mPieceList[i].generate_piece();
 46 	}
 47 	
 48 	document.getElementById( this.mPieceDiv ).innerHTML = pieceHtml;
 49 }

 51 /*
 52  * Get game piece surrounding coordinates
 53  * 
 54  * @param {coord} aCoord Selected coordinates
 55  * @return {GEPiece} Piece at coordinates
 56  */
 57 function get_piece_at_coord( aCoord )
 58 {
 59 	for( var i = 0; i < this.mPieceList.length; i++ )
 60 	{
 61 		var currentPiece = this.mPieceList[i];
 62 		//alert( i + " " + aCoord.left + " " + aCoord.top + " " + this.mPieceList[i].mRect.print() );
 63 		if( is_coord_in_rect( aCoord, currentPiece.mRect ) )
 64 		{
 65 			return currentPiece;
 66 		}
 67 	}
 68 	return null;
 69 }

 71 /**
 72  * Get piece at region
 73  *
 74  * @param {int} aRegionIdx Region Index
 75  * @return {GEPiece} Piece in region
 76  */
 77 function get_pieces_in_region( aRegionIdx )
 78 {
 79 	var ret = new Array();
 80 	
 81 	for( var pieceIdx = 0; pieceIdx < this.mPieceList.length; pieceIdx++ )
 82 	{
 83 		if( this.mPieceList[pieceIdx].get_current_region() == aRegionIdx )
 84 		{
 85 			ret[ret.length] = this.mPieceList[pieceIdx];
 86 		}
 87 	}
 88 	
 89 	return ret;
 90 }

 92 /**
 93  * Handle Mouse Move Event
 94  *
 95  * @param {event} event Mouse event
 96  */
 97 function handle_mouse_move(event)
 98 {	
 99 	if( this.mSelectedPiece )
100 	{
101 		// Get piece dimensions
102 		var pieceWidth = this.mSelectedPiece.mRect.w;
103 		var pieceHeight = this.mSelectedPiece.mRect.h;
104 		
105 		// Set piece coordinates
106 		var moveCoord = new coord(
107 							event.clientX - pieceWidth / 2,
108 							event.clientY - pieceHeight / 2
109 							);
110 							
111 		this.mSelectedPiece.move_piece_to_coords( moveCoord );
112 	}
113 }

115 /** 
116  * Handle Mouse Up Event
117  *
118  * @param {event} event Mouse Up Event
119  */
120 function handle_mouse_up
121 	(
122 	event
123 	)
124 {	
125 	// Bail if no piece selected
126 	if( !this.mSelectedPiece )
127 	{
128 		return;
129 	}
130 	
131 	// Validate move
132 	if( !gRulebook.validate_board() )
133 	{
134 		// Move is invalid. Revert to original position
135 		this.mSelectedPiece.move_piece_to_coords( this.mSelectedPieceStartCoord );
136 	}
137 	
138 	// Arrange pieces
139 	gBoard.cleanup_board();
140 	
141 	// Clear Selected Piece
142 	this.mSelectedPiece = null;
143 	this.mSelectedPieceStartCoord = new coord( 0, 0 );
144 }
145 
146 //
147 //  Handle Mouse Down Event
148 //
149 function handle_mouse_down(event)
150 {
151 	// Get mouse position
152 	var mouseCoord = new coord( event.clientX, event.clientY );
153 	
154 	// Determine selected piece
155 	this.mSelectedPiece = this.get_piece_at_coord( mouseCoord );
156 	
157 	// Store the current piece coordinate incase the 
158 	// attempted move is invalid
159 	if( this.mSelectedPiece )
160 	{
161 		this.mSelectedPieceStartCoord = new coord( 
162 												  this.mSelectedPiece.mRect.left, 
163 												  this.mSelectedPiece.mRect.top 
164 												  );
165 	}
166 }
